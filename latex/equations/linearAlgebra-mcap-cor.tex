\begin{equation}
	\left(
	\mathbf{M}_{k}
	\left(
	c
	\right)
	\right)_{ij, gh}
	=
	\begin{cases}
		\frac{1}{2}
		  &   
		\text{if}
		\quad
		\left( i, j \right) = \left( g, h \right)
		\text{or}
		\left( i, j \right) = \left( h, g \right),
		\quad
		i \neq j
		,
		g \neq h, \\
		0
		  &   
		\text{if}
		\quad
		i = j = g = h
		,
		\\
		0
		  &   
		\text{otherwise}
		.
	\end{cases}
	\label{eq:linearAlgebra-mcap-cor}
\end{equation}
