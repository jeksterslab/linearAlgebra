% Nel (1985) p. 141-142
\begin{equation}
	\left(
	\mathbf{M}_{k}
	\left(
	s
	\right)
	\right)_{ij, gh}
	=
	\begin{cases} 
		1 
		  &   
		\text{if} 
		\quad 
		i = j = g = h
		,
		\\
		\frac{1}{2} 
		  &   
		\text{if} 
		\quad 
		\left( i, j \right) = \left( g, h \right) 
		\text{or} 
		\left( i, j \right) = \left( h, g \right)
		, 
		\\
		0 
		  &   
		\text{otherwise}
		.
	\end{cases}
	\label{eq:linearAlgebra-mcap-sym}
\end{equation}
